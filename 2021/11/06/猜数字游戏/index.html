#include <stdio.h>
#include <time.h>
#include <stdlib.h>
void game()
{
int guess=0;
printf(“请输入一个数\n”);
int r=rand()%100+1;//生成一个1-100的随机数
while(1)
{
scanf(“%d”,&guess);
if(guess<r)
{
printf(“猜小了\n”);

    }
    else if(guess>r)
    {
      printf("猜大了\n");
    
    }
    else 
    {
    printf("恭喜你，猜对了\n");
    break;
    }    
}

}
menu()
{
printf(“******************\n”);
printf(“ 1.play \n”);
printf(“ 0.exit *****\n”);
printf(“***********************\n”);
}
int main()
{
int input=0;
srand((unsigned int)time(NULL));
do
{
menu();
printf(“请选择：> “);
scanf(“%d”,&input);
switch(input)
{
case(1):
printf(“玩游戏\n”);
game();
input = 0;
break;
case(0):
printf(“退出游戏\n”);
break;
default:
printf(“请输入1或2\n”);
break;
}
}
while(input);
return 0;
}
